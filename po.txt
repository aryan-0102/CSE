#include <bits/stdc++.h>
using namespace std;

int main() {
    vector<string> code = {
        "MOV R1, R1",
        "ADD R2, 0",
        "MUL R3, 2",
        "LOAD A",
        "LOAD A"
    };

    vector<string> opt;

    for (int i = 0; i < code.size(); i++) {
        string ins = code[i];

        if (ins == "MOV R1, R1")
            continue;

        if (ins == "ADD R2, 0")
            continue;

        if (ins == "MUL R3, 2") {
            opt.push_back("SHL R3, 1");
            continue;
        }

        if (i < code.size() - 1 && code[i] == code[i+1])
            continue;

        opt.push_back(ins);
    }

    cout << "Optimized Code:\n";
    for (auto &x : opt) cout << x << "\n";
}
